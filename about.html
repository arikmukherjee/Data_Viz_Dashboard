Features

Everything you need,
nothing you don't.

Every feature was chosen for signal clarity and developer ergonomics.

Three Chart Types
Bar, Line, and Doughnut charts powered by Chart.js 4 with gradient fills and smooth easing animations.

Live Data Simulation

Theme Engine
Dark and Light mode toggle, plus 4 accent color presets. Charts re-render instantly with matching gradients.

PNG Export

// Build user growth line chart
function buildLineChart() {
  const d = DATA[state.period];
  const ac = ACCENT_COLORS[state.accent];

  // Gradient area fill
  const grad = ctx.createLinearGradient(
    0, 0, 0, 280
  );
  grad.addColorStop(0, `rgba(${ac.rgb},0.18)`);
  grad.addColorStop(1, `rgba(${ac.rgb},0)`);

  state.charts.line = new Chart(ctx, {
    type: 'line',
    data: {
      labels: d.labels,
      datasets: [{
        data: d.users,
        borderColor: ac.hex,
        backgroundColor: grad,
        tension: 0.45
      }]
    }
  });
}
